GAS 全自動中文建表腳本

請在 Google 試算表中點選「擴充功能 > Apps Script」，將以下程式碼貼上並發布為網頁應用程式。
注意：此腳本會自動在你的 Google Drive 中建立「學生畫板上傳圖片」資料夾來存放學生的照片，並在試算表寫入連結。 


function doOptions(e) {
  return ContentService.createTextOutput("")
    .setMimeType(ContentService.MimeType.JSON)
    .setHeader("Access-Control-Allow-Origin", "*")
    .setHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS")
    .setHeader("Access-Control-Allow-Headers", "Content-Type");
}

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName("畫記紀錄");
    
    // 若工作表不存在則自動建立並寫入符合該情境的繁體中文欄位標題
    if (!sheet) {
      sheet = ss.insertSheet("畫記紀錄");
      sheet.appendRow(["提交時間", "學生姓名", "任務名稱", "圖片預覽網址"]);
      sheet.getRange("A1:D1").setFontWeight("bold").setBackground("#f3f4f6");
      sheet.setColumnWidth(4, 400); // 加寬圖片網址欄位
    }

    // 自動建立或尋找 Drive 資料夾存放圖片
    var folderName = "學生畫板上傳圖片";
    var folders = DriveApp.getFoldersByName(folderName);
    var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(folderName);
    
    // 解析 Base64 並存成圖片檔
    var base64Data = data.image.split(",")[1];
    var fileName = data.name + "_" + Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyyMMdd_HHmmss") + ".jpg";
    var blob = Utilities.newBlob(Utilities.base64Decode(base64Data), "image/jpeg", fileName);
    var file = folder.createFile(blob);
    
    // 設定檔案共用權限為知道連結者可檢視，方便前端讀取
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    
    // 獲取圖片直接檢視網址 (用於 html img src)
    var fileId = file.getId();
    var directUrl = "https://drive.google.com/uc?id=" + fileId;

    // 寫入試算表
    sheet.appendRow([
      Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy/MM/dd HH:mm:ss"),
      data.name,
      data.taskName,
      directUrl
    ]);

    return ContentService.createTextOutput(JSON.stringify({"status": "success", "url": directUrl}))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
      
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()}))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  }
}

function doGet(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("畫記紀錄");
    if(!sheet) return ContentService.createTextOutput("[]").setMimeType(ContentService.MimeType.JSON).setHeader("Access-Control-Allow-Origin", "*");
    
    var data = sheet.getDataRange().getValues();
    var result = [];
    // 倒序讀取，最新提交在最前
    for(var i = data.length - 1; i > 0; i--) {
      result.push({
        time: data[i][0],
        name: data[i][1],
        taskName: data[i][2],
        imageUrl: data[i][3]
      });
    }
    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({"error": err.toString()}))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  }
}
